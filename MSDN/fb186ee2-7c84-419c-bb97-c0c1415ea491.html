Не очень понятна суть вопроса. Можно указать имя файла при создании базы:

CREATE DATABASE MyBase ON 
( 
  NAME = MyBase_dat,
  FILENAME = 'C:\data\MyBase.mdf'
)
LOG ON
( 
  NAME = MyBase_log,
  FILENAME = 'C:\data\MyBaseLog.ldf'    
) ;

Можно отсоединить базу командой

EXEC sp_detach_db 'MyBase', 'true';

, переместить файлы, а затем снова присоединить базу:

CREATE DATABASE MyBase
      ON (FILENAME = 'D:\Data\MyNewBase.mdf') 
      FOR ATTACH ;
