Да тут все просто. Нужно перетащить на форму userTableAdapter (если его еще там нету), в нужном месте кода вызвать у него метод FillPass, и построить строку на основе содержимого DataTable. Как-то так:

this.userTableAdapter.FillPass(this.myDataSet.userTable,userid);

StringBuilder sb=new StringBuilder(300);
foreach (DataRow row in this.myDataSet.userTable.Rows)
{
    foreach (object item in row.ItemArray)
    {
        sb.Append(item.ToString()+"; "); 
    }
    sb.AppendLine();
}
textBox1.Text = sb.ToString();
