Если окно было открыто через JavaScript, так:

String csname1 = "MyScript";
Type cstype = this.GetType();

// Get a ClientScriptManager reference from the Page class.
ClientScriptManager cs = Page.ClientScript;

String cstext1 = "window.close();";
cs.RegisterStartupScript(cstype, csname1, cstext1, true);

если окно было открыто пользователем, скорее всего не выйдет, но можно попробовать так:

String csname1 = "MyScript";
Type cstype = this.GetType();

// Get a ClientScriptManager reference from the Page class.
ClientScriptManager cs = Page.ClientScript;

String cstext1 = "window.open('','_self').close();";
cs.RegisterStartupScript(cstype, csname1, cstext1, true);
