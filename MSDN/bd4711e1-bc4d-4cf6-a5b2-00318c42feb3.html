А как примерно будет изменён данный запрос с использованием ключей. Куда они тут будут вставлены. В общих чертах.

Например так:

System.Net.CredentialCache credentialCache = new System.Net.CredentialCache(); 
credentialCache.Add(
    new System.Uri(destinationUrl),
    "Basic", 
    new System.Net.NetworkCredential("login", "password")
);


httpWebRequest.Credentials = credentialCache;
